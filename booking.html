// Simple UI script for tier selection const cards = document.querySelectorAll('.tier-card'); const summaryName = document.getElementById('summary-tier-name'); const summaryTotal = document.getElementById('summary-total'); cards.forEach(card => { card.addEventListener('click', () => { // Remove active classes cards.forEach(c => { c.classList.remove('border-primary', 'bg-primary/5'); c.classList.add('border-white/5'); }); // Add active classes to clicked card.classList.add('border-primary', 'bg-primary/5'); card.classList.remove('border-white/5'); // Update summary summaryName.innerText = card.getAttribute('data-name'); summaryTotal.innerText = card.getAttribute('data-price') + ',00€'; }); }); // Check URL for plan (e.g. from Home page) const urlParams = new URLSearchParams(window.location.search); const tierParam = urlParams.get('tier'); if (tierParam) { // Mapping for simplicity const map = { '1-day': '1 Tag', '5-days': '5 Tage', '10-days': '10 Tage', 'unlimited': 'Unlimited' }; const targetName = map[tierParam]; const targetCard = Array.from(cards).find(c => c.getAttribute('data-name') === targetName); if (targetCard) targetCard.click(); }
